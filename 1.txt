# THEORY:
# Activation function is a mathematical function used in Artificial Neural Networks
# to decide whether a neuron should be activated or not.
# It introduces non-linearity into the network.

# Common Activation Functions:
# 1. Sigmoid  -> Output range (0 to 1)
# 2. Tanh     -> Output range (-1 to 1)
# 3. ReLU     -> Output range (0 to infinity)
# 4. Identity -> Output same as input
# 5. Softmax  -> Converts outputs into probabilities

# APPLICATIONS:
# - Used in classification problems
# - Helps neurons make decisions
# - Introduces non-linearity
# - Improves learning capability of neural networks

# Import NumPy library for mathematical operations
import numpy as np

# Import Matplotlib for plotting graph
import matplotlib.pyplot as plt

# Generate 100 equally spaced values from -5 to 5
x = np.linspace(-5, 5, 100)

# Plot Sigmoid Activation Function
# Formula: 1 / (1 + e^-x)
plt.plot(x, 1 / (1 + np.exp(-x)), label='Sigmoid')

# Plot Tanh Activation Function
# Formula: tanh(x)
plt.plot(x, np.tanh(x), label='Tanh')

# Plot ReLU Activation Function
# Formula: max(0, x)
plt.plot(x, np.maximum(0, x), label='ReLU')

# Plot Identity / Linear Activation Function
# Formula: f(x) = x
plt.plot(x, x, label='Identity')

# Plot Softmax Activation Function
# Formula: e^x / sum(e^x)
plt.plot(x, np.exp(x) / np.sum(np.exp(x)), label='Softmax')

# Label x-axis
plt.xlabel('Input')

# Label y-axis
plt.ylabel('Activation')

# Set graph title
plt.title('Activation Functions')

# Show legend for function names
plt.legend()

# Show grid
plt.grid(True)

# Display graph
plt.show()

# CONCLUSION

# ============================================================
# EXPERIMENT 1: Plot Activation Functions used in Neural Networks
# ============================================================

# AIM:
# Write a Python program to plot activation functions used in neural networks.

# OBJECTIVE:
# To learn activation functions and implement them using Python.

# THEORY:
# Activation Function:
# An activation function is a mathematical function used in Artificial Neural Networks
# to decide whether a neuron should be activated or not.
# It introduces non-linearity into the network.

# Why do we use Activation Functions?
# 1. To introduce non-linearity in the neural network.
# 2. To help the network learn complex patterns.
# 3. To normalize neuron output.
# 4. To improve model performance in classification tasks.

# Types of Activation Functions:

# 1. Sigmoid Function
# Formula: f(x) = 1 / (1 + e^-x)
# Output Range: 0 to 1
# Used in binary classification.

# 2. Tanh Function
# Formula: tanh(x)
# Output Range: -1 to 1
# Better than sigmoid in many hidden layers.

# 3. ReLU Function
# Formula: f(x) = max(0, x)
# Output Range: 0 to positive infinity
# Fast and commonly used in deep learning.

# 4. Softmax Function
# Converts outputs into probability distribution.
# Used in multiclass classification.

# APPLICATIONS:
# 1. Used in classification problems
# 2. Used in hidden layers of neural networks
# 3. Helps in decision making of neurons
# 4. Used in deep learning models

# ============================================================
# PROGRAM
# ============================================================

# Import NumPy for mathematical operations
import numpy as np

# Import Matplotlib for plotting graphs
import matplotlib.pyplot as plt

# Sigmoid activation function
def sigmoid(x):
    return 1 / (1 + np.exp(-x))

# Tanh activation function
def tanh(x):
    return np.tanh(x)

# ReLU activation function
def relu(x):
    return np.maximum(0, x)

# Softmax activation function
def softmax(x):
    exp_x = np.exp(x)
    return exp_x / np.sum(exp_x)

# Generate input values
x = np.linspace(-10, 10, 100)

# Apply activation functions
sigmoid_y = sigmoid(x)
tanh_y = tanh(x)
relu_y = relu(x)
softmax_y = softmax(x)

# Create figure window
plt.figure(figsize=(10, 8))

# Plot Sigmoid
plt.subplot(2, 2, 1)
plt.plot(x, sigmoid_y)
plt.title("Sigmoid Function")
plt.grid(True)

# Plot Tanh
plt.subplot(2, 2, 2)
plt.plot(x, tanh_y)
plt.title("Tanh Function")
plt.grid(True)

# Plot ReLU
plt.subplot(2, 2, 3)
plt.plot(x, relu_y)
plt.title("ReLU Function")
plt.grid(True)

# Plot Softmax
plt.subplot(2, 2, 4)
plt.plot(x, softmax_y)
plt.title("Softmax Function")
plt.grid(True)

# Adjust spacing
plt.tight_layout()

# Display plots
plt.show()

# Test values
test = np.array([-1, 0, 1])

# Print outputs
print("Sigmoid:", sigmoid(test))
print("Tanh:", tanh(test))
print("ReLU:", relu(test))
print("Softmax:", softmax(test))
